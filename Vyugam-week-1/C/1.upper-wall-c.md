# 1.Upper Wall - EASY PROBLEM - Solution in C 


```c

#include<stdio.h>
int main ()
{              char s[100000];
               int i;
               gets(s); 
               for( i=0 ; s[i]!='\0' ; i++)
               {               if ( s[i]== ' ' )
                                printf( "\n" );
                                else
                                printf(  "%c",  s[i]-32);            
               }
                  return 0;
}
```
